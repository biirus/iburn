### Views

```jsx
  <Button>
    Default Button
  </Button>

  <Button view="thin">
    Thin Button
  </Button>

  <Button view="link">
    Link Button
  </Button>
```

### Sizes

```jsx
  <Button size="inline" color="Primary">
    Primary inline Button
  </Button>

  <Button size="compact" color="help">
    Help compact Button
  </Button>

  <Button view="block" color="warning">
    Warning block Button
  </Button>
```

### Colors

```jsx
<Button>
  Default Button
</Button>

<Button color="primary">
  Primary Button
</Button>

<Button color="warning">
  Warning Button
</Button>

<Button color="danger">
  Danger Button
</Button>

<Button color="help">
  Help Button
</Button>
```
